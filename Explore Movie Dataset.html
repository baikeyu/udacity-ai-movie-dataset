def select_type(x):
    if "Comedy" in x and "Drama" in x:
        return "Comedy & Drama"
    elif "Comedy" in x and "Drama" not in x:
        return "Comedy"
    elif "Comedy" not in x and "Drama" in x:
        return "Drama"
    else:
        return "Other"
